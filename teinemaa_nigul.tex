\section*{Defensive and Competitive Bidding}
\smallsection{Overcalls}
	\nth{1} level (4)5\more cards, 8-16 \hcp \rightarrow \sfrac{1}{1}\caps{resp}=F1; \sfrac21\caps{resp}=\caps{NF}\\
	\quad fit-jumps, \caps{oppt} suit=\caps{inv} to 4M;\\
	\nth{2} level 5\more cards, 11-17 \hcp \rightarrow new suit=F1\\ 
	\caps{Trf-Leb} after: {\smaller(1M/P)-\X/1\D/1\N-(2\D M)-?, (1m/M/P)-(2M)-\X-?}:\\
	\quad2y=\caps{nf}; 2\N=any weak or clubs; 3\C\D\H=\TRF\ \caps{inv\more}; \\
	\quad3\S=\caps{Trf}\ to 3\N; \TRF\ to \caps{oppt} suit=\caps{Stay} \GF\\
	\caps{Trf} after 2\C-(2M): 2\N/3\C=\C; 3\S=\D\GF\\
	\TRF\ after (1m/Pass)–Dbl/1NT–(2\C/3\C) and (2\C/3\C)–\X\\
	\caps{Leb} after 2\D-(2M): 2\N=\caps{Leb}; 3\C/\D=\caps{inv\more}/\caps{nf}; 3\H/\S=\H/\C\GF\\
\smallsection{1\N\ Overcall} 
	\VUL15-18, \NV 14-17, \sfrac12\more \caps{stop}\rightarrow \TRF s; \TRF\ to \caps{oppt} M=\caps{Stay}\\
	Reopen: 1\N=11-15, 0+\caps{stop}; 2\N=19-21 
\smallsection{Jump Overcalls} 
	Weak\rightarrow  new suit=\INV; 2\N=relay\\
	2\N=two lowest suits; Over 1m=\caps{nat}: 2\D=5\more4\more MM, 8-11. \\
	Reopen: intermediate
\smallsection{Direct \& Jump Cue Bids} 
	Over 1M: 2M=oM+m, 10\more\rightarrow 2\N=relay; 3\C=P/C; 3\D=\INV\ M.\\
	\quad 3M=ask stopper for 3\N, usually with running suit\\
	Over 2\C: 3\C=MM; 4\C=\D+\H
\smallsection{Vs NT} 
	\caps{Asptro}: \X=(14)16\more; 2\H\S=6 cards, 8-15; \\
	\quad 2\C/\D=4-5\H/\S\ \caps{unbal}, 10-15, (if MM, we bid shorter first);\\
	% \sout{\quad 2\N=6-5 same rank/color, \INV\more}\\
	Passed hand: \X=5+m\\
	Over (1M)-(1\N): direct \X=\takeout; 2M=oM+m;
	\ Reopen: \NAT
\smallsection{Vs Preempts} 
	\X=\takeout\ thru 4\H, cards higher. \CUE=ask stopper.\\
	\caps{Trf-Leb} after (2x)-\X. (2M)-4m=m+oM. 3\N-4\C=(R) w/\caps{bal}\\
	Over Multi 2\D: \X=\takeout\ of \S; 2\N=16-19; 4m=\H+m
\smallsection{Vs Artificial Strong Openings} 
	Over 1m, 1\C-1\D: \X=5-4 MM, 10\more; 1\N=5-4 mm\\
	\quad2\N=good hand w/ one-suiter or minors\\
	Over 2m: \X=MM; 2\N=mm
\smallsection{Over Opponents' Takeout Double}
	Over 1\C-(\X): \XX=\NAT\ \GF; 1\D=5-8\rightarrow\NAT\ continuations\\
	Over 1\D: \TRF\ from \XX=\H;\\
	Over 1M: \XX=10+\hcp, F2M; \TRF\ from 1\N=\C; 3x=fit-jump\\
	Over 1\N(14-17)-(\X): system on\\
